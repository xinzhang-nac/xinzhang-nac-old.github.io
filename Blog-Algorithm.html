Union Find

# Union Find 写作框架
## setup
parents = [i for i in range(n)]
ranks = [0 for i in range(n)]
## define find func and union func
def find(x):
  if x != parents[x]:
      parents[x] = find(parents[x])
  return parents[x]
def union(x,y):
  px, py = find(x), find(y)
    if ranks[px] > ranks[py]: 
      parents[py] = px
    if ranks[px] < ranks[py]: 
      parents[px] = py
    if ranks[px] == ranks[py]: 
      parents[py] = px
      ranks[px] += 1
